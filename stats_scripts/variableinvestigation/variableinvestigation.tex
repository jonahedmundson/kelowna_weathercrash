+   for (j in c(1:12)){

+       select = - c(`Temp.Flag`, 
+       `Dew.Point.Temp.Flag`, `Rel.Hum.Flag`,

+       `Wind.Spd.Flag`, `Visibility.Flag`, 
+       `Stn.Press.Flag`, `Hmdx`, `Hmdx.Flag`, `Wind.Chill.Flag`))
+     weatherdata = rbind(weatherdata, temp)
+   }
+ }
\end{Sinput}
